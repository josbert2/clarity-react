## Usage

```tsx
import { UserIcon } from "lucide-react"

import {
  Tabs,
  TabsContent,
  TabsList,
  TabsTrigger,
  TabsTriggerIcon,
  TabsTriggerText,
} from "@/components/annui/focus-tabs"
```

```tsx
<Tabs defaultValue="account">
  <TabsList>
    <TabsTrigger value="account">
      <TabsTriggerIcon>
        <UserIcon />
      </TabsTriggerIcon>
      <TabsTriggerText>Account</TabsTriggerText>
    </TabsTrigger>
  </TabsList>
  <TabsContent value="account">This is the account content.</TabsContent>
</Tabs>
```
